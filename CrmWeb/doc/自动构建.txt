ant脚本文件：


<?xml version="1.0" encoding="GBK"?>
<project basedir="." default="all" name="VsopWeb">
	<path id="project.class.path">
		<fileset dir="${project.dir}/WebRoot/WEB-INF/lib">
			<include name="*.jar"/>
		</fileset> 
		<fileset dir="${project.dir}/WebRoot/WEB-INF/classes">
			<include name="**/*.*"/>
		</fileset> 
	</path>
	
	<target name="init">
		<mkdir dir="${project.dir}/build"/>
		<delete failonerror="false" includeEmptyDirs="true">
			<fileset dir="${project.dir}/WebRoot/WEB-INF/classes"/>
		</delete>
		<mkdir dir="${project.dir}/WebRoot/WEB-INF/classes"/>
	</target>
	
	
	
	<target name="buildVsopWeb" depends="">
		<mkdir dir="${project.dir}/WebRoot/WEB-INF/classes"/>
		<javac destdir="${project.dir}/WebRoot/WEB-INF/classes" executable="/root/bea/jdk150_12/bin/javac" compiler="javac1.4" memoryInitialSize="128m" memoryMaximumSize="512m" fork="true" failonerror="true" debug="true">
			<src path="${project.dir}/src" /> 
			<classpath refid="project.class.path"/>
		</javac>
		
		<copy todir="${project.dir}/WebRoot/WEB-INF/classes">
			<fileset dir="${project.dir}/src">
				<include name="*.properties"/>
				<include name="*.xml"/>
			</fileset>
		</copy>
		
		<delete failonerror="false" includeEmptyDirs="true">
			<fileset dir="${project.dir}/build/VsopWeb.war"/>
			<fileset dir="${deploy.dir}/VsopWeb.war"/>
		</delete>
		
		<war compress="false" destfile="${project.dir}/build/VsopWeb.war" webxml="${project.dir}/WebRoot/WEB-INF/web.xml">
			<zipfileset dir="${project.dir}/WebRoot"></zipfileset>
		</war>
		
		<copy todir="${deploy.dir}">
			<fileset dir="${project.dir}/build/">
				<include name="*.war"/>
			</fileset>
		</copy>
		
	</target>		
	
	<target name="all" depends="init,buildVsopWeb">
	</target>
	
</project>
